package main

import (
	"slices"
)


func kidsWithCandies(candies []int, extraCandies int) []bool {
	var output []bool
	for i := range candies {
		if candies[i]+extraCandies >= slices.Max(candies) {
			output = append(output, true)
		} else {
			output = append(output, false)
		}
	}
	return output
}

func canPlaceFlowers(flowerbed []int, n int) bool {
	count := 0
	length := len(flowerbed)

	for i := 0; i < length; i++ {
		// Проверяем, что текущая позиция свободна
		if flowerbed[i] == 0 {
			// Проверяем, что слева свободно (или это начало)
			left := (i == 0) || (flowerbed[i-1] == 0)
			// Проверяем, что справа свободно (или это конец)
			right := (i == length-1) || (flowerbed[i+1] == 0)

			if left && right {
				// Сажаем цветок
				flowerbed[i] = 1
				count++
				// Если посадили достаточно — можно выходить
				if count >= n {
					return true
				}
			}
		}
	}

	return count >= n
}

  package main

import "strings"

func reverseVowels(s string) string {
	vowels := "aeiouAEIOUаеёиоуыэюяАЕЁИОУЫЭЮЯ"
	runeSlice := []rune(s)

	// Собираем все гласные в порядке их появления
	var vowelList []rune
	for _, r := range runeSlice {
		if strings.ContainsRune(vowels, r) {
			vowelList = append(vowelList, r)
		}
	}

	// Переворачиваем список гласных
	for i, j := 0, len(vowelList)-1; i < j; i, j = i+1, j-1 {
		vowelList[i], vowelList[j] = vowelList[j], vowelList[i]
	}

	// Заменяем гласные в исходной строке на перевёрнутые
	vowelIndex := 0
	for i, r := range runeSlice {
		if strings.ContainsRune(vowels, r) {
			runeSlice[i] = vowelList[vowelIndex]
			vowelIndex++
		}
	}

	return string(runeSlice)
}

package main

import (
	"strings"
)

func reverseWords(s string) string {
	s_slice := strings.Fields(s)

	for i, j := 0, len(s_slice)-1; i < j; i, j = i+1, j-1 {
		s_slice[i], s_slice[j] = s_slice[j], s_slice[i]
	}

	return strings.Join(s_slice, " ")
}
