 ### Хардовая задача еще раз перерешать!
```go
func compress(chars []byte) int {
    write := 0   // сюда будем писать результат (в начало массива)
    i := 0       // текущая позиция в массиве

    for i < len(chars) {
        currentChar := chars[i]
        count := 0

        // Считаем, сколько раз подряд идёт currentChar
        for i < len(chars) && chars[i] == currentChar {
            i++
            count++
        }

        // Шаг 1: записываем сам символ
        chars[write] = currentChar
        write++

        // Шаг 2: если повторов > 1 — записываем цифры числа
        if count > 1 {
            countStr := strconv.Itoa(count)
            for j := 0; j < len(countStr); j++ {
                chars[write] = countStr[j]
                write++
            }
        }
    }

    return write
}
```

func moveZeroes(nums []int) {
    write := 0  // куда писать следующий ненулевой элемент

    // Проходим по всему массиву
    for read := 0; read < len(nums); read++ {
        if nums[read] != 0 {
            nums[write] = nums[read]
            write++
        }
    }

    // Заполняем оставшиеся ячейки нулями
    for write < len(nums) {
        nums[write] = 0
        write++
    }
}

func isSubsequence(s string, t string) bool {
    i := 0 // указатель для s
    j := 0 // указатель для t

    for i < len(s) && j < len(t) {
        if s[i] == t[j] {
            i++ // нашли совпадение — двигаемся в s
        }
        j++ // всегда двигаемся в t
    }

    return i == len(s) // если прошли всю s — значит, это подпоследовательность
}
