```go
func productExceptSelf(nums []int) []int {
    n := len(nums)
    result := make([]int, n)

    // Шаг 1: заполняем result произведениями слева
    result[0] = 1
    for i := 1; i < n; i++ {
        result[i] = result[i-1] * nums[i-1]
    }

    // Шаг 2: идём справа, накапливая произведение справа (suffix)
    suffix := 1
    for i := n - 1; i >= 0; i-- {
        result[i] = result[i] * suffix
        suffix = suffix * nums[i]
    }

    return result
}
```

```go
func increasingTriplet(nums []int) bool {
    first := math.MaxInt32
    second := math.MaxInt32

    for _, num := range nums {
        if num <= first {
            first = num
        } else if num <= second {
            second = num
        } else {
            return true
        }
    }

    return false
}
```
